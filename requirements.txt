# --- Core y Utilidades ---
python-dotenv   # Carga variables de entorno desde archivos .env para la configuración segura.
joblib          # Serializa y deserializa objetos de Python; optimiza el guardado de modelos.

# --- Ciencia de Datos y Machine Learning ---
pandas          # Manipula y analiza estructuras de datos estructurados (DataFrames).
scikit-learn    # Provee herramientas y algoritmos para minería de datos y aprendizaje automático.
shap            # Explica las predicciones de los modelos de ML mediante valores SHAP.

# --- Visualización y Dashboards ---
matplotlib      # Genera gráficos estáticos básicos y visualizaciones de datos en 2D.
seaborn         # Crea gráficos estadísticos atractivos basados en Matplotlib.
plotly          # Habilita gráficos interactivos y dinámicos para la web.
streamlit       # Construye aplicaciones web de datos y dashboards interactivos rápidamente.

# --- API y Backend (FastAPI) ---
fastapi         # Construye APIs modernas y de alto rendimiento basadas en estándares.
uvicorn         # Ejecuta la aplicación como un servidor ASGI asíncrono y ligero.
pydantic        # Valida datos y gestiona ajustes mediante anotaciones de tipos de Python.
bcrypt          # Hashea contraseñas de forma segura para la autenticación de usuarios.
requests        # Peticiones desde el Backend - Frontend

# --- Base de Datos ---
sqlalchemy      # Gestiona la base de datos mediante un ORM (Mapeo Objeto-Relacional).
psycopg2-binary # Conecta Python con bases de datos PostgreSQL (adaptador binario).
